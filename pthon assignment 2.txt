class Counter:
    def _init_(self, value):
        self.value = value

    def increase_one(self):
        self.value = self.value + 1
        return self.value
    def decrease_one(self):
        self.value = self.value - 1
        return self.value
    def get_value(self):
        print(self.value)


x = Counter(1)

x.get_value()
x.increase_one()
x.increase_one()
x.get_value()
x.decrease_one()
x.decrease_one()
x.get_value()

class Person:
    def _init_(self, args):
        self.name = args[0]
        self.lastname = args[1]
        self.dateofbirth = args[2]

    def printdetails(self):
        print(f"Name :{self.name}")
        print(f"Last Name :{self.lastname}")
        print(f"Date of birth :{self.dateofbirth}")

    def is_eligible(self, age):
        if age >= 18:
            return True
        else:
            return False


x = Person(["BOB", "DYLAN", "24 MAY 1941"])

x.printdetails()
age = int(input())
print(x.is_eligible(age))

    if x % 3 != 0 and x % 5 != 0:

def prime_numbers(x):
    for j in range(2,x+1):
        count = 0
        for i in range(2,(j//2+1)):
            if j%i==0:
                count = 1
                break
        if count == 0:
            print(j,end = ' ')
if _name_ == "_main_":
    x = int(input())
    prime_numbers(x)
